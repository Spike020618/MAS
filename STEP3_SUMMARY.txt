💡 使用示例
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

from mas.rag import (
    LocalRAGDatabase,
    RAGWorkflow,
    RAGSyncManager,
    MultiAgentCoordinator
)

# 初始化
rag_db = LocalRAGDatabase()
workflow = RAGWorkflow(rag_db)
sync_mgr = RAGSyncManager(agent_id=0, agent_name="Coordinator")
coordinator = MultiAgentCoordinator(0, "Coordinator", workflow, sync_mgr)

# 执行多Agent任务分配
result = await coordinator.allocate_task_with_sync(
    task_request={"task_id": "...", "description": "..."},
    enable_remote=True
)
